#include<stdio.h>
#include<assert.h>
#include<string.h>
int count_str(const char*s1, const char*s2)
{
	assert(s1&&s2);
	char*keep = (char*)s1;
	char*p1 = NULL;
	char*p2 = NULL;
	int count = 0;
	while (*keep!='\n')
	{
		p1 = keep;
		p2 = (char*)s2;
		while (*p1&&*p2 && !(*p1 - *p2))
		{
			p1++;
			p2++;
		}
		if (!(*p2))
			count++;
		keep++;
	}
	return count;
}
int main()
{
	int i = 0;
	char p[8000] = { 0 };
	char q[] = "CHN";
	while ((p[i] = getchar())!= '\n')
	{
		i++;
	}
	printf("%d", count_str(p, q));
}
